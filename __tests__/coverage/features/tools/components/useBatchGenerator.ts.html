import { useState, useCallback } from 'react';
import { useTranslation } from 'react-i18next';
import { useUIStore } from '../../../store/useUIStore';
import { generateFunctionByDifficulty, type FunctionType } from '../../../utils/mathEngine/generator';
import type { ChapterData, LevelData } from '../../../types/story';
import { BatchGeneratorConfigSchema } from '../../../utils/schemas';
import { GAME_CONSTANTS } from '../../../utils/constants';
 
export interface ChapterConfig {
  id: string;
  name: string;
  levelCount: number;
  startDifficulty: number;
  endDifficulty: number;
  withParams: boolean;
  allowedTypes: FunctionType[];
}
 
export const ALL_TYPES: { value: FunctionType; label: string; zhLabel: string }[] = [
  { value: 'polynomial', label: 'Polynomial (x, x^2)', zhLabel: '多项式 (x, x^2)' },
  { value: 'absolute', label: 'Absolute (|x|)', zhLabel: '绝对值 (|x|)' },
  { value: 'rational', label: 'Rational (1/x)', zhLabel: '有理分式 (1/x)' },
  { value: 'radical', label: 'Radical (sqrt)', zhLabel: '无理根式 (sqrt)' },
  { value: 'exponential', label: 'Exponential (e^{x}, ln)', zhLabel: '指数/对数 (e^x, ln)' },
  { value: 'trigonometric', label: 'Trigonometric (sin, cos)', zhLabel: '三角函数 (sin, cos)' },
  { value: 'inverse_trigonometric', label: 'Inv Trig (arcsin, arctan)', zhLabel: '反三角函数 (arcsin)' },
  { value: 'hyperbolic', label: 'Hyperbolic (sinh, cosh)', zhLabel: '双曲函数 (sinh, cosh)' },
  { value: 'inverse_hyperbolic', label: 'Inv Hyperbolic (arsinh)', zhLabel: '反双曲函数 (arsinh)' }
];
 
export function useBatchGenerator(onGenerate: (chapters: ChapterData[]) => void, onClose: () => void) {
  const { t } = useTranslation();
  
  const [configs, setConfigs] = useState<ChapterConfig[]>([
    {
      id: 'ch_random_1',
      name: t('tools.storyEditor.newChapterTitle', { id: 1, defaultValue: `Chapter 1` }),
      levelCount: 5,
      startDifficulty: GAME_CONSTANTS.GENERATOR.MIN_DIFFICULTY,
      endDifficulty: GAME_CONSTANTS.GENERATOR.MAX_DIFFICULTY,
      withParams: false,
      allowedTypes: ['polynomial', 'absolute']
    }
  ]);
 
  const handleAddChapter = useCallback(() => {
    setConfigs(prev => {
      const prevChapter = prev.length > 0 ? prev[prev.length - 1] : null;
      const inheritedStart = prevChapter ? prevChapter.endDifficulty : GAME_CONSTANTS.GENERATOR.MIN_DIFFICULTY;
      const inheritedEnd = prevChapter ? prevChapter.endDifficulty : GAME_CONSTANTS.GENERATOR.MIN_DIFFICULTY + 2;
      
      return [
        ...prev,
        {
          id: `ch_random_${prev.length + 1}`,
          name: t('tools.storyEditor.newChapterTitle', { id: prev.length + 1, defaultValue: `Chapter ${prev.length + 1}` }),
          levelCount: 5,
          startDifficulty: inheritedStart,
          endDifficulty: inheritedEnd,
          withParams: prevChapter ? prevChapter.withParams : false,
          allowedTypes: prevChapter ? [...prevChapter.allowedTypes] : ['polynomial']
        }
      ];
    });
  }, [t]);
 
  const handleRemoveChapter = useCallback((index: number) => {
    setConfigs(prev => prev.filter((_, i) => i !== index));
  }, []);
 
  const updateConfig = useCallback(<K extends keyof ChapterConfig>(index: number, field: K, value: ChapterConfig[K]) => {
    setConfigs(prev => {
      const next = [...prev];
      next[index] = { ...next[index], [field]: value };
      return next;
    });
  }, []);
 
  const toggleType = useCallback((index: number, type: FunctionType) => {
    setConfigs(prev => {
      const next = [...prev];
      const types = new Set(next[index].allowedTypes);
      if (types.has(type)) {
        types.delete(type);
      } else {
        types.add(type);
      }
      next[index] = { ...next[index], allowedTypes: Array.from(types) };
      return next;
    });
  }, []);
 
  const handleGenerate = useCallback(() => {
    let globalLevelCounter = 1;
    
    const generatedChapters: ChapterData[] = configs.map(config => {
      const levels: LevelData[] = [];
      
      const FLUCTUATION_RANGE = 0.6;
      const WITH_PARAMS_PROB = 0.6;
      const TARGET_DIFF_PROB = 0.3;
 
      for (let i = 0; i < config.levelCount; i++) {
        const progress = config.levelCount > 1 ? i / (config.levelCount - 1) : 0;
        const baseDiff = config.startDifficulty + progress * (config.endDifficulty - config.startDifficulty);
        
        const fluctuation = (Math.random() * FLUCTUATION_RANGE) - (FLUCTUATION_RANGE / 2);
        const diff = Math.max(GAME_CONSTANTS.GENERATOR.MIN_DIFFICULTY, Math.min(GAME_CONSTANTS.GENERATOR.MAX_DIFFICULTY, baseDiff + fluctuation));
        
        const wp = config.withParams && Math.random() < WITH_PARAMS_PROB;
        const targetDiff = (Math.random() < TARGET_DIFF_PROB) ? Math.max(0, diff - 1) : diff;
 
        const generated = generateFunctionByDifficulty({
          targetDifficulty: targetDiff,
          withParams: wp,
          allowedTypes: config.allowedTypes.length > 0 ? config.allowedTypes : undefined
        });
 
        levels.push({
          id: `${globalLevelCounter}`,
          title: t('tools.storyEditor.newLevel', { id: globalLevelCounter, defaultValue: `Level ${globalLevelCounter}` }),
          targetFunction: generated.target,
          params: Object.keys(generated.params).length > 0 ? generated.params : null,
          domain: null,
          type: 'normal',
          unlockConditions: null,
          tip: null
        });
        
        globalLevelCounter++;
      }
 
      return {
        id: config.id,
        title: config.name,
        levels,
        files: []
      };
    });
 
    onGenerate(generatedChapters);
    onClose();
  }, [configs, onGenerate, onClose, t]);
 
  const handleExport = useCallback(() => {
    const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(configs, null, 2));
    const downloadAnchorNode = document.createElement('a');
    downloadAnchorNode.setAttribute("href", dataStr);
    downloadAnchorNode.setAttribute("download", "batch-config.json");
    document.body.appendChild(downloadAnchorNode);
    downloadAnchorNode.click();
    downloadAnchorNode.remove();
  }, [configs]);
 
  const handleImport = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (file) {
      const reader = new FileReader();
      reader.onload = (event) => {
        try {
          const json = JSON.parse(event.target?.result as string);
          
          // 使用 Zod 进行严格的 schema 校验
          const result = BatchGeneratorConfigSchema.safeParse(json);
          if (result.success) {
            setConfigs(result.data as ChapterConfig[]);
          } else {
            console.error("Zod validation failed:", result.error);
            useUIStore.getState().openDialog({ type: 'alert', message: t('tools.storyEditor.parseError') });
          }
        } catch (err: unknown) {
          useUIStore.getState().openDialog({ type: 'alert', message: t('tools.storyEditor.parseError') });
          console.error(err);
        }
      };
      reader.readAsText(file);
    }
    e.target.value = '';
  }, [t]);
 
  return {
    configs,
    handleAddChapter,
    handleRemoveChapter,
    updateConfig,
    toggleType,
    handleGenerate,
    handleExport,
    handleImport
  };
}
