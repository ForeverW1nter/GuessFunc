import React, { Suspense } from 'react';
import { useUIStore } from '../../../store/useUIStore';
 
const SettingsModal = React.lazy(() => import('./SettingsModal').then(m => ({ default: m.SettingsModal })));
const LevelSelectModal = React.lazy(() => import('../../story/components/LevelSelectModal').then(m => ({ default: m.LevelSelectModal })));
const RandomChallengeModal = React.lazy(() => import('./RandomChallengeModal').then(m => ({ default: m.RandomChallengeModal })));
 
export const GlobalModals = () => {
  const isSettingsOpen = useUIStore(state => state.isSettingsOpen);
  const isLevelSelectOpen = useUIStore(state => state.isLevelSelectOpen);
  const isRandomChallengeOpen = useUIStore(state => state.isRandomChallengeOpen);
 
  return (
    <Suspense fallback={null}>
      {isSettingsOpen && <SettingsModal />}
      {isLevelSelectOpen && <LevelSelectModal />}
      {isRandomChallengeOpen && <RandomChallengeModal />}
    </Suspense>
  );
};
